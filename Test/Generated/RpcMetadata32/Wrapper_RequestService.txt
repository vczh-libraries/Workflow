func rpc_IRpcObjectOps(lc : system::IRpcLifeCycle*) : (system::IRpcObjectOps^)
{
    return new (system::IRpcObjectOps^)
    {
        var _lc : system::IRpcLifeCycle* = lc;

        var _services : system::Interface^[int] = {};

        delete
        {
            for (service in _services.Values)
            {
                var wrapper : system::IRpcWrapperBase^ = (service as (system::IRpcWrapperBase^));
                if (wrapper is not null)
                {
                    wrapper.DisconnectFromLifecycle();
                }
            }
        }

        override func InvokeMethod(ref : system::RpcObjectReference, methodId : int, arguments : system::Array^) : (object)
        {
            switch (methodId)
            {
                case rpcmethod_RpcTest__IService_GetText:
                {
                    var target : RpcTest::IService^ = (cast (RpcTest::IService^) _lc.RefToPtr(ref));
                    return system::IRpcLifeCycle::RpcBoxByval(target.GetText(), _lc);
                }
                default:
                raise "Unknown RPC method id.";
            }
        }

        override func InvokeMethodAsync(ref : system::RpcObjectReference, methodId : int, arguments : system::Array^) : (system::Async^)
        {
            raise "InvokeMethodAsync is not supported.";
        }

        override func ObjectHold(ref : system::RpcObjectReference, remoteClientId : int, hold : bool) : (void)
        {
            if (hold)
            {
                _lc.Controller.AcquireRemoteObject(ref);
            }
            else
            {
                _lc.Controller.ReleaseRemoteObject(ref);
            }
        }

        override func RegisterService(typeId : int, service : system::Interface^) : (void)
        {
            {
                if (rpcwrapper_IsCtorInterfaceTypeId(typeId))
                {
                    if (service is null)
                    {
                        _services.Remove(typeId);
                    }
                    else
                    {
                        _services.Set(typeId, service);
                    }
                }
                else
                {
                    if (service is not null)
                    {
                        if (rpcwrapper_IsInterfaceTypeId(typeId))
                        {
                            raise "RPC service type id is not an @rpc:Ctor interface.";
                        }
                        else
                        {
                            raise "RPC service type id does not exist.";
                        }
                    }
                }
            }
        }

        override func RequestService(typeId : int) : (system::RpcObjectReference)
        {
            {
                if (_services.Keys.Contains(typeId))
                {
                    return _lc.PtrToRef(_services[typeId]);
                }
                return {};
            }
        }
    };
}
