module Rpc;
using system::*;

namespace RpcInheritance
{
  @rpc:Interface
  interface IValue
  {
    @rpc:Cached
    prop Value : string {const}
  }

  @rpc:Interface
  interface IOne : IValue
  {
    event CrashAtServer();
    func SetOneValue() : void;
  }

  @rpc:Interface
  interface ITwo : IValue
  {
    event CrashAtClient();
    func SetTwoValue() : void;
  }

  @rpc:Interface
  interface IDerived : IOne, ITwo
  {
    func SetDerivedValue() : void;
    func GuardCrashAtClient() : string;
  }

  @rpc:Interface
  @rpc:Ctor
  interface IService
  {
    func CreateOne() : IOne^;
    func CreateTwo() : ITwo^;
    func CreateDerived() : IDerived^;
  }
}
