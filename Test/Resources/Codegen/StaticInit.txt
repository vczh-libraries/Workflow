module test;
using system::*;
using test::*;

var map : int[string] = {};

namespace A
{
	namespace B
	{
		static
		{
			map.Set("A::B", 0);
		}
	}

	static
	{
		map.Set("A(0)", 0);
	}

	namespace C
	{
		static
		{
			map.Set("A::C", 0);
		}
	}

	static
	{
		map.Set("A(1)", 0);
	}
}

func main():string
{
	var result = "";
	for (key in map.Keys)
	{
		if (result != "")
		{
			result = result & ", ";
		}
		result = result & (cast string key);
	}
	return result;
}

static
{
	map.Set("::", 0);
}
