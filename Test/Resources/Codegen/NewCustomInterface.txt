module test;
using system::*;
using test::*;

interface IMyInterface : Enumerable
{
	static func Get50() : int
	{
		return 50;
	}

	func Get100() : int;
}

func main():string
{
	var begin = 1;
	var end = 5;
	var xs = new IMyInterface^
	{
		func CreateEnumerator() : Enumerator^
		{
			return new Enumerator^
			{
				var index = -1;

				func GetCurrent() : object
				{
					return begin + index;
				}

				func GetIndex() : int
				{
					return index;
				}

				func Next() : bool
				{
					if(index == end - begin)
					{
						return false;
					}
					else
					{
						index = index + 1;
						return true;
					}
				}
			};
		}

		func Get100() : int
		{
			return Get50() * 2;
		}
	};
	return $"[$(xs.Get100())][$(CustomInterfaceProcessor::Sum(xs))]";
}
