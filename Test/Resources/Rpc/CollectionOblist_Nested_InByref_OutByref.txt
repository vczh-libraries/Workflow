module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollectionOblist
{
  namespace Nested
  {
    namespace InByref
    {
      namespace OutByref
      {
        @rpc:Interface
        @rpc:Ctor
        interface IService
        {
          @rpc:Byref
          func DoList(@rpc:Byref xs : observe (observe int[])[]) : observe (observe int[])[];
        }
      }
    }
  }
}

var xsService : observe (observe int[])[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
  var serviceObj = new (RpcCollectionOblist::Nested::InByref::OutByref::IService^)
  {
    override func DoList(xs : observe (observe int[])[]) : observe (observe int[])[]
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is null))
      {
        raise "Parameter xs should be a wrapper object in serviceMain";
      }

      xsService = xs;
      var ys = xs[0];
      ys.Add(4);
      return xs;
    }
  };
  lc.RegisterService("RpcCollectionOblist::Nested::InByref::OutByref::IService", serviceObj);
}

func Print(xs : observe (observe int[])[]) : string
{
  var result = "";
  for (x in xs[0])
  {
    result = $"$(result)$(x)";
  }
  return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
  var xsOrigin : observe (observe int[])[] = {{1 2 3}};
  var service = cast (RpcCollectionOblist::Nested::InByref::OutByref::IService^) lc.RequestService("RpcCollectionOblist::Nested::InByref::OutByref::IService");
  var xs = service.DoList(xsOrigin);

  if (xs != xsOrigin)
  {
    raise "Return value xs should be the original parameter object in clientMain";
  }

  var xsClient = xs;
  var ys = xs[0];
  ys.Add(5);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
