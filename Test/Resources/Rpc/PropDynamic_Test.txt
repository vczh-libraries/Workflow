module Rpc;
using system::*;
using RpcWrapperTest::*;


var s = "";

func ClientValueChanged() : void
{
  s = $"$(s)[ValueChanged]";
}

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (RpcPropDynamic::IService^)
  {
    var _Value : string = "A";

    override func GetValue() : string
    {
      return _Value;
    }

    override func SetValue(value : string) : void
    {
      _Value = value;
    }

    override func Signal() : void
    {
      ValueChanged();
    }
  };
  lc.RegisterService("RpcPropDynamic::IService", serviceObj);
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var service = cast (RpcPropDynamic::IService^) lc.RequestService("RpcPropDynamic::IService");
  attach(service.ValueChanged, ClientValueChanged);

  s = $"$(s)[$(service.Value)]";

  service.SetValue("B");
  s = $"$(s)[$(service.Value)]";

  service.Signal();
  s = $"$(s)[$(service.Value)]";

  return s;
}
