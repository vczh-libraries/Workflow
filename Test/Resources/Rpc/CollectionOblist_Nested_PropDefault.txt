module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollection
{
	namespace Nested
	{
		namespace PropDefault
		{
			@rpc:Interface
			@rpc:Ctor
			interface IService
			{
				prop List : (int[])[] {not observe}
			}
		}
	}
}

var xsService : (int[])[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (RpcCollection::Nested::PropDefault::IService^)
	{
		var _List : (int[])[] = null;

		override func GetList() : (int[])[]
		{
			return _List;
		}

		override func SetList(xs : (int[])[]) : void
		{
			var xsObj : object = xs;
			if ((xsObj as (system::IRpcWrapperBase^) is not null))
			{
				raise "Parameter xs should be a copied local object in serviceMain";
			}

			xsService = xs;
			var ys = xs[0];
			ys.Add(4);
			_List = xs;
		}
	};
	lc.RegisterService("RpcCollection::Nested::PropDefault::IService", serviceObj);
}

func Print(xs : (int[])[]) : string
{
	var result = "";
	for (x in xs[0])
	{
		result = $"$(result)$(x)";
	}
	return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin = {{1 2 3}};
	var service = cast (RpcCollection::Nested::PropDefault::IService^) lc.RequestService("RpcCollection::Nested::PropDefault::IService");
	service.List = xsOrigin;
	var xs = service.List;

	if (xs == xsOrigin)
	{
		raise "Return value xs should be a copied local object in clientMain";
	}

	var xsObj : object = xs;

	if ((xsObj as (system::IRpcWrapperBase^) is not null))
	{
		raise "Return value xs should be a copied local object in clientMain";
	}

	var xsClient = xs;
	var ys = xs[0];
	ys.Add(5);

	return
		$"[$(Print(xsOrigin))]" &
		$"[$(Print(xsService))]" &
		$"[$(Print(xsClient))]";
}
