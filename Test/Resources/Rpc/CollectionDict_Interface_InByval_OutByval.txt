module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcCollectionDict::Interface::InByval::OutByval::*;

namespace RpcCollectionDict
{
	namespace Interface
	{
		namespace InByval
		{
			namespace OutByval
			{
				@rpc:Interface
				interface IValue
				{
					func GetValue() : string;
					prop Value : string {GetValue}
				}

				@rpc:Interface
				@rpc:Ctor
				interface IService
				{
					@rpc:Byval
					func DoList(@rpc:Byval xs : IValue^[int]) : IValue^[int];
				}
			}
		}
	}
}

var xsService : IValue^[int] = {};

func MakeValue(value : int) : IValue^
{
	return MakeValue($"$(value)");
}

func MakeValue(value : string) : IValue^
{
	return new (IValue^)
	{
		override func GetValue() : string
		{
			return value;
		}
	};
}

func CheckValue(value : IValue^, shouldBeWrapper : bool, name : string) : void
{
	var valueObj : object = value;
	if (shouldBeWrapper)
	{
		if ((valueObj as (system::IRpcWrapperBase^) is null))
		{
			raise $"$(name) should be a wrapper object";
		}
	}
	else
	{
		if ((valueObj as (system::IRpcWrapperBase^) is not null))
		{
			raise $"$(name) should be a local object";
		}
	}
}

func CheckServiceItems(xs : IValue^[int]) : void
{
	for (i in range [1, xs.Count])
	{
	  CheckValue(xs[i], i != 4, $"xsService[$(i)]");
	}
}

func CheckClientItems(xs : IValue^[int]) : void
{
	for (i in range [1, xs.Count])
	{
	  CheckValue(xs[i], i == 4, $"xsService[$(i)]");
	}
}

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (IService^)
	{
		override func DoList(xs : IValue^[int]) : IValue^[int]
		{
			var xsObj : object = xs;
			if ((xsObj as (system::IRpcWrapperBase^) is not null))
			{
				raise "Parameter xs should be a copied local object in serviceMain";
			}

			xsService = xs;
			xs.Set(4, MakeValue("D"));
			return xs;
		}
	};
	lc.RegisterService("RpcCollectionDict::Interface::InByval::OutByval::IService", serviceObj);
}

func Print(xs : IValue^[int]) : string
{
	var result = "";
	for (key in range [1, xs.Count])
	{
		result = $"$(result)$(key)$(xs[key].Value)";
	}
	return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin : IValue^[int] = {};
	xsOrigin.Set(1, MakeValue("A"));
	xsOrigin.Set(2, MakeValue("B"));
	xsOrigin.Set(3, MakeValue("C"));
	var service = cast (IService^) lc.RequestService("RpcCollectionDict::Interface::InByval::OutByval::IService");
	var xs = service.DoList(xsOrigin);
	var xsObj : object = xs;

	if (xs == xsOrigin)
	{
		raise "Return value xs should be a copied local object in clientMain";
	}

	if ((xsObj as (system::IRpcWrapperBase^) is not null))
	{
		raise "Return value xs should be a copied local object in clientMain";
	}

	var xsClient = xs;
	xs.Set(5, MakeValue("E"));

	CheckServiceItems(xsService);
	CheckClientItems(xsClient);

	return
		$"[$(Print(xsOrigin))]" &
		$"[$(Print(xsService))]" &
		$"[$(Print(xsClient))]";
}
