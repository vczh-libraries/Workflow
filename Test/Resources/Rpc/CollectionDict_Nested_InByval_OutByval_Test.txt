module Rpc;
using system::*;
using RpcWrapperTest::*;


var xsService : (string[int])[] = {};

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (RpcCollectionDict::Nested::InByval::OutByval::IService^)
  {
    override func DoList(xs : (string[int])[]) : (string[int])[]
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is not null))
      {
        raise "Parameter xs should be a copied local object in serviceMain";
      }

      xsService = xs;
      var ys = xs[0];
      ys.Set(4, "D");
      return xs;
    }
  };
  lc.RegisterLocalService(rpc_GetIds()["RpcCollectionDict::Nested::InByval::OutByval::IService"], serviceObj);
}

func Print(xs : (string[int])[]) : string
{
  var ys = xs[0];
  var result = "";
  for (key in range [1, ys.Count])
  {
    result = $"$(result)$(key)$(ys[key])";
  }
  return result;
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var xsOrigin = {{1:"A" 2:"B" 3:"C"}};
  var service = cast (RpcCollectionDict::Nested::InByval::OutByval::IService^) lc.RequestService("RpcCollectionDict::Nested::InByval::OutByval::IService");
  var xs = service.DoList(xsOrigin);
  var xsObj : object = xs;

  if (xs == xsOrigin)
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  if ((xsObj as (system::IRpcWrapperBase^) is not null))
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  var xsClient = xs;
  var ys = xs[0];
  ys.Set(5, "E");

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
