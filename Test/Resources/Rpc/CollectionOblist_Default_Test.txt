module Rpc;
using system::*;
using RpcWrapperTest::*;


var xsService : observe int[] = {};

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (RpcCollectionOblist::Default::IService^)
  {
    override func DoList(xs : observe int[]) : observe int[]
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is null))
      {
        raise "Parameter xs should be a wrapper object in serviceMain";
      }

      xsService = xs;
      xs.Add(4);
      return xs;
    }
  };
  lc.RegisterLocalService(rpc_GetIds()["RpcCollectionOblist::Default::IService"], serviceObj);
}

func Print(xs : observe int[]) : string
{
  var result = "";
  for (x in xs)
  {
    result = $"$(result)$(x)";
  }
  return result;
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var xsOrigin : observe int[] = {1 2 3};
  var service = cast (RpcCollectionOblist::Default::IService^) lc.RequestService("RpcCollectionOblist::Default::IService");
  var xs = service.DoList(xsOrigin);

  if (xs != xsOrigin)
  {
    raise "Return value xs should be the original parameter object in clientMain";
  }

  var xsClient = xs;
  xs.Add(5);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
