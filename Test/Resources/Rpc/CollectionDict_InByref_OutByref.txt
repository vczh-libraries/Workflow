module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcCollectionDict::InByref::OutByref::*;

namespace RpcCollectionDict
{
	namespace InByref
	{
		namespace OutByref
		{
			@rpc:Interface
			@rpc:Ctor
			interface IService
			{
				@rpc:Byref
				func DoList(@rpc:Byref xs : string[int]) : string[int];
			}
		}
	}
}

var xsService : string[int] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (IService^)
	{
		override func DoList(xs : string[int]) : string[int]
		{
			var xsObj : object = xs;
			if ((xsObj as (system::IRpcWrapperBase^) is null))
			{
				raise "Parameter xs should be a wrapper object in serviceMain";
			}

			xsService = xs;
			xs.Set(4, "D");
			return xs;
		}
	};
	lc.RegisterService("RpcCollectionDict::InByref::OutByref::IService", serviceObj);
}

func Print(xs : string[int]) : string
{
	var result = "";
	for (key in range [1, xs.Count])
	{
		result = $"$(result)$(key)$(xs[key])";
	}
	return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin = {1:"A" 2:"B" 3:"C"};
	var service = cast (IService^) lc.RequestService("RpcCollectionDict::InByref::OutByref::IService");
	var xs = service.DoList(xsOrigin);

	if (xs != xsOrigin)
	{
		raise "Return value xs should be the original parameter object in clientMain";
	}

	var xsClient = xs;
	xs.Set(5, "E");

	return
		$"[$(Print(xsOrigin))]" &
		$"[$(Print(xsService))]" &
		$"[$(Print(xsClient))]";
}
