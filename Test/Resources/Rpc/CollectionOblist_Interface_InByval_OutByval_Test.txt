module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcCollectionOblist::Interface::InByval::OutByval::*;


var xsService : observe IValue^[] = {};

func MakeValue(value : int) : IValue^
{
  return MakeValue($"$(value)");
}

func MakeValue(value : string) : IValue^
{
  return new (IValue^)
  {
    override func GetValue() : string
    {
      return value;
    }
  };
}

func CheckValue(value : IValue^, shouldBeWrapper : bool, name : string) : void
{
  var valueObj : object = value;
  if (shouldBeWrapper)
  {
    if ((valueObj as (system::IRpcWrapperBase^) is null))
    {
      raise $"$(name) should be a wrapper object";
    }
  }
  else
  {
    if ((valueObj as (system::IRpcWrapperBase^) is not null))
    {
      raise $"$(name) should be a local object";
    }
  }
}

func CheckServiceItems(xs : observe IValue^[]) : void
{
  for (i in range [0, xs.Count))
  {
    CheckValue(xs[i], i != 3, $"xsService[$(i)]");
  }
}

func CheckClientItems(xs : observe IValue^[]) : void
{
  for (i in range [0, xs.Count))
  {
    CheckValue(xs[i], i == 3, $"xsService[$(i)]");
  }
}

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (IService^)
  {
    override func DoList(xs : observe IValue^[]) : observe IValue^[]
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is not null))
      {
        raise "Parameter xs should be a copied local object in serviceMain";
      }

      xsService = xs;
      xs.Add(MakeValue(4));
      return xs;
    }
  };
  lc.RegisterLocalService(rpc_GetIds()["RpcCollectionOblist::Interface::InByval::OutByval::IService"], serviceObj);
}

func Print(xs : observe IValue^[]) : string
{
  var result = "";
  for (x in xs)
  {
    result = $"$(result)$(x.Value)";
  }
  return result;
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var xsOrigin : observe IValue^[] = {};
  xsOrigin.Add(MakeValue(1));
  xsOrigin.Add(MakeValue(2));
  xsOrigin.Add(MakeValue(3));
  var service = cast (IService^) lc.RequestService("RpcCollectionOblist::Interface::InByval::OutByval::IService");
  var xs = service.DoList(xsOrigin);
  var xsObj : object = xs;

  if (xs == xsOrigin)
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  if ((xsObj as (system::IRpcWrapperBase^) is not null))
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  var xsClient = xs;
  xs.Add(MakeValue(5));

  CheckServiceItems(xsService);
  CheckClientItems(xsClient);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
