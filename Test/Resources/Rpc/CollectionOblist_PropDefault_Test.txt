module Rpc;
using system::*;
using RpcWrapperTest::*;


var xsService : observe int[] = {};

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (RpcCollectionOblist::PropDefault::IService^)
  {
    var _List : observe int[] = null;

    override func GetList() : observe int[]
    {
      return _List;
    }

    override func SetList(xs : observe int[]) : void
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is null))
      {
        raise "Parameter xs should be a wrapper object in serviceMain";
      }

      xsService = xs;
      xs.Add(4);
      _List = xs;
    }
  };
  lc.RegisterService("RpcCollectionOblist::PropDefault::IService", serviceObj);
}

func Print(xs : observe int[]) : string
{
  var result = "";
  for (x in xs)
  {
    result = $"$(result)$(x)";
  }
  return result;
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var xsOrigin : observe int[] = {1 2 3};
  var service = cast (RpcCollectionOblist::PropDefault::IService^) lc.RequestService("RpcCollectionOblist::PropDefault::IService");
  service.List = xsOrigin;
  var xs = service.List;

  if (xs != xsOrigin)
  {
    raise "Return value xs should be the original parameter object in clientMain";
  }

  var xsClient = xs;
  xs.Add(5);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
