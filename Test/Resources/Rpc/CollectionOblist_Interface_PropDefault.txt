module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcCollectionOblist::Interface::PropDefault::*;

namespace RpcCollectionOblist
{
  namespace Interface
  {
    namespace PropDefault
    {
      @rpc:Interface
      interface IValue
      {
        func GetValue() : string;
        @rpc:Dynamic
        prop Value : string {GetValue}
      }

      @rpc:Interface
      @rpc:Ctor
      interface IService
      {
        @rpc:Dynamic
        prop List : observe IValue^[] {not observe}
      }
    }
  }
}

var xsService : observe IValue^[] = {};

func MakeValue(value : int) : IValue^
{
  return MakeValue($"$(value)");
}

func MakeValue(value : string) : IValue^
{
  return new (IValue^)
  {
    override func GetValue() : string
    {
      return value;
    }
  };
}

func CheckValue(value : IValue^, shouldBeWrapper : bool, name : string) : void
{
  var valueObj : object = value;
  if (shouldBeWrapper)
  {
    if ((valueObj as (system::IRpcWrapperBase^) is null))
    {
      raise $"$(name) should be a wrapper object";
    }
  }
  else
  {
    if ((valueObj as (system::IRpcWrapperBase^) is not null))
    {
      raise $"$(name) should be a local object";
    }
  }
}

func CheckServiceItems(xs : observe IValue^[]) : void
{
  for (i in range [0, xs.Count))
  {
    CheckValue(xs[i], i != 3, $"xsService[$(i)]");
  }
}

func CheckClientItems(xs : observe IValue^[]) : void
{
  for (i in range [0, xs.Count))
  {
    CheckValue(xs[i], i == 3, $"xsService[$(i)]");
  }
}

func serviceMain(lc : IRpcLifeCycle*) : void
{
  var serviceObj = new (IService^)
  {
    var _List : observe IValue^[] = null;

    override func GetList() : observe IValue^[]
    {
      return _List;
    }

    override func SetList(xs : observe IValue^[]) : void
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is null))
      {
        raise "Parameter xs should be a wrapper object in serviceMain";
      }

      xsService = xs;
      xs.Add(MakeValue(4));
      _List = xs;
    }
  };
  lc.RegisterService("RpcCollectionOblist::Interface::PropDefault::IService", serviceObj);
}

func Print(xs : observe IValue^[]) : string
{
  var result = "";
  for (x in xs)
  {
    result = $"$(result)$(x.Value)";
  }
  return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
  var xsOrigin : observe IValue^[] = {};
  xsOrigin.Add(MakeValue(1));
  xsOrigin.Add(MakeValue(2));
  xsOrigin.Add(MakeValue(3));
  var service = cast (IService^) lc.RequestService("RpcCollectionOblist::Interface::PropDefault::IService");
  service.List = xsOrigin;
  var xs = service.List;

  if (xs != xsOrigin)
  {
    raise "Return value xs should be the original parameter object in clientMain";
  }

  var xsClient = xs;
  xs.Add(MakeValue(5));

  CheckServiceItems(xsService);
  CheckClientItems(xsClient);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
