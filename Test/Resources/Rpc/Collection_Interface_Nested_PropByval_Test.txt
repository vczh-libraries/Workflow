module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcCollection::Interface::Nested::InByval::PropByval::*;


var xsService : IValue^[][int] = {};

func MakeValue(value : int) : IValue^
{
  return new (IValue^)
  {
    override func GetValue() : int
    {
      return value;
    }
  };
}

func CheckValue(value : IValue^, shouldBeWrapper : bool, name : string) : void
{
  var valueObj : object = value;
  if (shouldBeWrapper)
  {
    if ((valueObj as (system::IRpcWrapperBase^) is null))
    {
      raise $"$(name) should be a wrapper object";
    }
  }
  else
  {
    if ((valueObj as (system::IRpcWrapperBase^) is not null))
    {
      raise $"$(name) should be a local object";
    }
  }
}

func CheckServiceItems(xs : IValue^[][int]) : void
{
  var ys = xs[0];
  for (i in range [0, ys.Count))
  {
    CheckValue(ys[i], i != 3, $"xsService[0][$(i)]");
  }
}

func CheckClientItems(xs : IValue^[][int]) : void
{
  var ys = xs[0];
  for (i in range [0, ys.Count))
  {
    CheckValue(ys[i], i == 3, $"xsClient[0][$(i)]");
  }
}

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (IService^)
  {
    var _List : IValue^[][int] = null;

    override func GetList() : IValue^[][int]
    {
      return _List;
    }

    override func SetList(xs : IValue^[][int]) : void
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is not null))
      {
        raise "Parameter xs should be a copied local object in serviceMain";
      }

      xsService = xs;
      var ys = xs[0];
      ys.Add(MakeValue(4));
      _List = xs;
    }
  };
  lc.RegisterService("RpcCollection::Interface::Nested::InByval::PropByval::IService", serviceObj);
}

func Print(xs : IValue^[][int]) : string
{
  var result = "";
  for (x in xs[0])
  {
    result = $"$(result)$(x.Value)";
  }
  return result;
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var ysOrigin : IValue^[] = {};
  ysOrigin.Add(MakeValue(1));
  ysOrigin.Add(MakeValue(2));
  ysOrigin.Add(MakeValue(3));
  var xsOrigin : IValue^[][int] = {};
  xsOrigin.Set(0, ysOrigin);
  var service = cast (IService^) lc.RequestService("RpcCollection::Interface::Nested::InByval::PropByval::IService");
  service.List = xsOrigin;
  var xs = service.List;
  var xsObj : object = xs;

  if (xs == xsOrigin)
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  if ((xsObj as (system::IRpcWrapperBase^) is not null))
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  var xsClient = xs;
  var ys = xs[0];
  ys.Add(MakeValue(5));

  CheckServiceItems(xsService);
  CheckClientItems(xsClient);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
