module Rpc;
using system::*;

namespace RpcInheritanceEventException
{
  @rpc:Interface
  interface IValue
  {
  }

  @rpc:Interface
  interface IOne : IValue
  {
    event CrashAtServer();
  }

  @rpc:Interface
  interface ITwo : IValue
  {
    event CrashAtClient();
  }

  @rpc:Interface
  interface IDerived : IOne, ITwo
  {
    func GuardCrashAtClient() : string;
  }

  @rpc:Interface
  @rpc:Ctor
  interface IService
  {
    func CreateDerived() : IDerived^;
  }
}
