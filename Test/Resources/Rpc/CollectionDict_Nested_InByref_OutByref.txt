module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollectionDict
{
	namespace Nested
	{
		namespace InByref
		{
			namespace OutByref
			{
				@rpc:Interface
				@rpc:Ctor
				interface IService
				{
					@rpc:Byref
					func DoList(@rpc:Byref xs : (string[int])[]) : (string[int])[];
				}
			}
		}
	}
}

var xsService : (string[int])[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (RpcCollectionDict::Nested::InByref::OutByref::IService^)
	{
		override func DoList(xs : (string[int])[]) : (string[int])[]
		{
			xsService = xs;
			var ys = xs[0];
			ys.Set(4, "D");
			return xs;
		}
	};
	lc.RegisterService("RpcCollectionDict::Nested::InByref::OutByref::IService", serviceObj);
}

func Print(xs : (string[int])[]) : string
{
	var ys = xs[0];
	var result = "";
	for (key in range [1, ys.Count])
	{
		result = $"$(result)$(key)$(ys[key])";
	}
	return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin = {{1:"A" 2:"B" 3:"C"}};
	var service = cast (RpcCollectionDict::Nested::InByref::OutByref::IService^) lc.RequestService("RpcCollectionDict::Nested::InByref::OutByref::IService");
	var xs = service.DoList(xsOrigin);

	if (xs != xsOrigin)
	{
		raise "Return value xs should be the original parameter object in clientMain";
	}

	var xsClient = xs;
	var ys = xs[0];
	ys.Set(5, "E");

	return
		$"[$(Print(xsOrigin))]" &
		$"[$(Print(xsService))]" &
		$"[$(Print(xsClient))]";
}
