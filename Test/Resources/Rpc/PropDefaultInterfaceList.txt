module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcPropDefaultInterfaceList::*;

namespace RpcPropDefaultInterfaceList
{
  @rpc:Interface
  interface IValue
  {
    func GetValue() : string;
    @rpc:Dynamic
    prop Value : string {GetValue}
  }

  @rpc:Interface
  @rpc:Ctor
  interface IService
  {
    prop Value : IValue^[] {GetValue : ValueChanged}
    func GetValue() : IValue^[];

    func SetValue(value : IValue^[]) : void;
    func Signal() : void;
    event ValueChanged();
  }
}
