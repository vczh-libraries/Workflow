module Rpc;
using system::*;
using RpcWrapperTest::*;


var xsService : (int[])[] = {};

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (RpcCollection::Nested::InByref::OutByref::IService^)
  {
    override func DoList(xs : (int[])[]) : (int[])[]
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is null))
      {
        raise "Parameter xs should be a wrapper object in serviceMain";
      }

      xsService = xs;
      var ys = xs[0];
      ys.Add(4);
      return xs;
    }
  };
  lc.RegisterService("RpcCollection::Nested::InByref::OutByref::IService", serviceObj);
}

func Print(xs : (int[])[]) : string
{
  var result = "";
  for (x in xs[0])
  {
    result = $"$(result)$(x)";
  }
  return result;
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var xsOrigin = {{1 2 3}};
  var service = cast (RpcCollection::Nested::InByref::OutByref::IService^) lc.RequestService("RpcCollection::Nested::InByref::OutByref::IService");
  var xs = service.DoList(xsOrigin);

  if (xs != xsOrigin)
  {
    raise "Return value xs should be the original parameter object in clientMain";
  }

  var xsClient = xs;
  var ys = xs[0];
  ys.Add(5);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
