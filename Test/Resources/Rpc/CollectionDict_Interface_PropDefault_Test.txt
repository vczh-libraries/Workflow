module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcCollectionDict::Interface::PropDefault::*;


var xsService : IValue^[int] = {};

func MakeValue(value : int) : IValue^
{
  return MakeValue($"$(value)");
}

func MakeValue(value : string) : IValue^
{
  return new (IValue^)
  {
    override func GetValue() : string
    {
      return value;
    }
  };
}

func CheckValue(value : IValue^, shouldBeWrapper : bool, name : string) : void
{
  var valueObj : object = value;
  if (shouldBeWrapper)
  {
    if ((valueObj as (system::IRpcWrapperBase^) is null))
    {
      raise $"$(name) should be a wrapper object";
    }
  }
  else
  {
    if ((valueObj as (system::IRpcWrapperBase^) is not null))
    {
      raise $"$(name) should be a local object";
    }
  }
}

func CheckServiceItems(xs : IValue^[int]) : void
{
  for (i in range [1, xs.Count])
  {
    CheckValue(xs[i], i != 4, $"xsService[$(i)]");
  }
}

func CheckClientItems(xs : IValue^[int]) : void
{
  for (i in range [1, xs.Count])
  {
    CheckValue(xs[i], i == 4, $"xsService[$(i)]");
  }
}

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (IService^)
  {
    var _List : IValue^[int] = null;

    override func GetList() : IValue^[int]
    {
      return _List;
    }

    override func SetList(xs : IValue^[int]) : void
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is null))
      {
        raise "Parameter xs should be a wrapper object in serviceMain";
      }

      xsService = xs;
      xs.Set(4, MakeValue("D"));
      _List = xs;
    }
  };
  lc.RegisterService("RpcCollectionDict::Interface::PropDefault::IService", serviceObj);
}

func Print(xs : IValue^[int]) : string
{
  var result = "";
  for (key in range [1, xs.Count])
  {
    result = $"$(result)$(key)$(xs[key].Value)";
  }
  return result;
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var xsOrigin : IValue^[int] = {};
  xsOrigin.Set(1, MakeValue("A"));
  xsOrigin.Set(2, MakeValue("B"));
  xsOrigin.Set(3, MakeValue("C"));
  var service = cast (IService^) lc.RequestService("RpcCollectionDict::Interface::PropDefault::IService");
  service.List = xsOrigin;
  var xs = service.List;

  if (xs != xsOrigin)
  {
    raise "Return value xs should be the original parameter object in clientMain";
  }

  var xsClient = xs;
  xs.Set(5, MakeValue("E"));

  CheckServiceItems(xsService);
  CheckClientItems(xsClient);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
