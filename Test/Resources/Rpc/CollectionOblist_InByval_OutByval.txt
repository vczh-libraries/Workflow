module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollectionOblist
{
  namespace InByval
  {
    namespace OutByval
    {
      @rpc:Interface
      @rpc:Ctor
      interface IService
      {
        @rpc:Byval
        func DoList(@rpc:Byval xs : observe int[]) : observe int[];
      }
    }
  }
}

var xsService : observe int[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
  var serviceObj = new (RpcCollectionOblist::InByval::OutByval::IService^)
  {
    override func DoList(xs : observe int[]) : observe int[]
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is not null))
      {
        raise "Parameter xs should be a copied local object in serviceMain";
      }

      xsService = xs;
      xs.Add(4);
      return xs;
    }
  };
  lc.RegisterService("RpcCollectionOblist::InByval::OutByval::IService", serviceObj);
}

func Print(xs : observe int[]) : string
{
  var result = "";
  for (x in xs)
  {
    result = $"$(result)$(x)";
  }
  return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
  var xsOrigin : observe int[] = {1 2 3};
  var service = cast (RpcCollectionOblist::InByval::OutByval::IService^) lc.RequestService("RpcCollectionOblist::InByval::OutByval::IService");
  var xs = service.DoList(xsOrigin);
  var xsObj : object = xs;

  if (xs == xsOrigin)
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  if ((xsObj as (system::IRpcWrapperBase^) is not null))
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  var xsClient = xs;
  xs.Add(5);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
