module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollectionOblist
{
  namespace Nested
  {
    namespace InByval
    {
      namespace PropByval
      {
        @rpc:Interface
        @rpc:Ctor
        interface IService
        {
          @rpc:Byval
          @rpc:Dynamic
          prop List : observe (observe int[])[] {not observe}
        }
      }
    }
  }
}

var xsService : observe (observe int[])[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
  var serviceObj = new (RpcCollectionOblist::Nested::InByval::PropByval::IService^)
  {
    var _List : observe (observe int[])[] = null;

    override func GetList() : observe (observe int[])[]
    {
      return _List;
    }

    override func SetList(xs : observe (observe int[])[]) : void
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is not null))
      {
        raise "Parameter xs should be a copied local object in serviceMain";
      }

      xsService = xs;
      var ys = xs[0];
      ys.Add(4);
      _List = xs;
    }
  };
  lc.RegisterService("RpcCollectionOblist::Nested::InByval::PropByval::IService", serviceObj);
}

func Print(xs : observe (observe int[])[]) : string
{
  var result = "";
  for (x in xs[0])
  {
    result = $"$(result)$(x)";
  }
  return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
  var xsOrigin : observe (observe int[])[] = {{1 2 3}};
  var service = cast (RpcCollectionOblist::Nested::InByval::PropByval::IService^) lc.RequestService("RpcCollectionOblist::Nested::InByval::PropByval::IService");
  service.List = xsOrigin;
  var xs = service.List;

  if (xs == xsOrigin)
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  var xsObj : object = xs;

  if ((xsObj as (system::IRpcWrapperBase^) is not null))
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  var xsClient = xs;
  var ys = xs[0];
  ys.Add(5);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
