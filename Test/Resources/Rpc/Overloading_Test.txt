module Rpc;
using system::*;
using RpcOverloadingTest::*;


var s = "";

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (RpcOverloadingTest::IService^)
  {
    override func ToStringInt(value : int) : string
    {
      return $"$(value)";
    }

    override func ToString(value : bool) : string
    {
      return $"$(value)";
    }

    override func ToString(value : string) : string
    {
      return value;
    }

    override func ToString(value : IStringRepresentable^) : string
    {
      return value.StringValue;
    }

    override func ToString(value1 : int, value2 : bool, value3 : string, value4 : IStringRepresentable^) : string
    {
      return $"$(value1),$(value2),$(value3),$(value4.StringValue)";
    }
  };
  lc.RegisterLocalService(rpc_GetIds()["RpcOverloadingTest::IService"], serviceObj);
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var service = cast (RpcOverloadingTest::IService^) lc.RequestService("RpcOverloadingTest::IService");
  var value1 = 123;
  var value2 = true;
  var value3 = "abc";
  var value4 = new (RpcOverloadingTest::IStringRepresentable^)
  {
    override func GetStringValue() : string
    {
      return "xyz";
    }
  };

  s = $"[$(service.ToStringInt(value1))]";
  s = $"$(s)[$(service.ToString(value2))]";
  s = $"$(s)[$(service.ToString(value3))]";
  s = $"$(s)[$(service.ToString(value4))]";
  s = $"$(s)[$(service.ToString(value1, value2, value3, value4))]";
  return s;
}
