module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcCollection::Interface::InByval::OutByref::*;

namespace RpcCollection
{
	namespace Interface
	{
		namespace InByval
		{
			namespace OutByref
			{
				@rpc:Interface
				interface IValue
				{
					func GetValue() : string;
					prop Value : string {GetValue}
				}

				@rpc:Interface
				@rpc:Ctor
				interface IService
				{
					@rpc:Byref
					func DoList(@rpc:Byval xs : IValue^[]) : IValue^[];
				}
			}
		}
	}
}

var xsService : IValue^[] = {};

func MakeValue(value : int) : IValue^
{
	return MakeValue($"$(value)");
}

func MakeValue(value : string) : IValue^
{
	return new (IValue^)
	{
		override func GetValue() : string
		{
			return value;
		}
	};
}

func CheckValue(value : IValue^, shouldBeWrapper : bool, name : string) : void
{
	var valueObj : object = value;
	if (shouldBeWrapper)
	{
		if ((valueObj as (system::IRpcWrapperBase^) is null))
		{
			raise $"$(name) should be a wrapper object";
		}
	}
	else
	{
		if ((valueObj as (system::IRpcWrapperBase^) is not null))
		{
			raise $"$(name) should be a local object";
		}
	}
}

func CheckServiceItems(xs : IValue^[]) : void
{
	CheckValue(xs[0], true, "xsService[0]");
	CheckValue(xs[1], true, "xsService[1]");
	CheckValue(xs[2], true, "xsService[2]");
	CheckValue(xs[3], false, "xsService[3]");
}

func CheckClientItems(xs : IValue^[]) : void
{
	CheckValue(xs[0], false, "xsClient[0]");
	CheckValue(xs[1], false, "xsClient[1]");
	CheckValue(xs[2], false, "xsClient[2]");
	CheckValue(xs[3], true, "xsClient[3]");
	CheckValue(xs[4], false, "xsClient[4]");
}

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (IService^)
	{
		override func DoList(xs : IValue^[]) : IValue^[]
		{
			var xsObj : object = xs;
			if ((xsObj as (system::IRpcWrapperBase^) is not null))
			{
				raise "Parameter xs should be a copied local object in serviceMain";
			}

			xsService = xs;
			xs.Add(MakeValue(4));
			return xs;
		}
	};
	lc.RegisterService("RpcCollection::Interface::InByval::OutByref::IService", serviceObj);
}

func Print(xs : IValue^[]) : string
{
	var result = "";
	for (x in xs)
	{
		result = $"$(result)$(x.Value)";
	}
	return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin : IValue^[] = {};
	xsOrigin.Add(MakeValue(1));
	xsOrigin.Add(MakeValue(2));
	xsOrigin.Add(MakeValue(3));
	var service = cast (IService^) lc.RequestService("RpcCollection::Interface::InByval::OutByref::IService");
	var xs = service.DoList(xsOrigin);
	var xsObj : object = xs;

	if ((xsObj as (system::IRpcWrapperBase^) is null))
	{
		raise "Return value xs should be a wrapper object in clientMain";
	}

	var xsClient = xs;
	xs.Add(MakeValue(5));

	CheckServiceItems(xsService);
	CheckClientItems(xsClient);

	return
		$"[$(Print(xsOrigin))]" &
		$"[$(Print(xsService))]" &
		$"[$(Print(xsClient))]";
}
