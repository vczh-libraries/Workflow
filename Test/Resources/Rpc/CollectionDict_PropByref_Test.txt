module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcCollectionDict::InByref::PropByref::*;


var xsService : string[int] = {};

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (IService^)
  {
    var _List : string[int] = null;

    override func GetList() : string[int]
    {
      return _List;
    }

    override func SetList(xs : string[int]) : void
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is null))
      {
        raise "Parameter xs should be a wrapper object in serviceMain";
      }

      xsService = xs;
      xs.Set(4, "D");
      _List = xs;
    }
  };
  lc.RegisterLocalService(rpc_GetIds()["RpcCollectionDict::InByref::PropByref::IService"], serviceObj);
}

func Print(xs : string[int]) : string
{
  var result = "";
  for (key in range [1, xs.Count])
  {
    result = $"$(result)$(key)$(xs[key])";
  }
  return result;
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var xsOrigin = {1:"A" 2:"B" 3:"C"};
  var service = cast (IService^) lc.RequestService("RpcCollectionDict::InByref::PropByref::IService");
  service.List = xsOrigin;
  var xs = service.List;

  if (xs != xsOrigin)
  {
    raise "Return value xs should be the original parameter object in clientMain";
  }

  var xsClient = xs;
  xs.Set(5, "E");

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
