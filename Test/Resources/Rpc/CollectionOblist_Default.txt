module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollectionOblist
{
	namespace Default
	{
		@rpc:Interface
		@rpc:Ctor
		interface IService
		{
			func DoList(xs : int[]) : int[];
		}
	}
}

var xsService : int[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (RpcCollectionOblist::Default::IService^)
	{
		override func DoList(xs : int[]) : int[]
		{
			var xsObj : object = xs;
			if ((xsObj as (system::IRpcWrapperBase^) is not null))
			{
				raise "Parameter xs should be a copied local object in serviceMain";
			}

			xsService = xs;
			xs.Add(4);
			return xs;
		}
	};
	lc.RegisterService("RpcCollectionOblist::Default::IService", serviceObj);
}

func Print(xs : int[]) : string
{
	var result = "";
	for (x in xs)
	{
		result = $"$(result)$(x)";
	}
	return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin = {1 2 3};
	var service = cast (RpcCollectionOblist::Default::IService^) lc.RequestService("RpcCollectionOblist::Default::IService");
	var xs = service.DoList(xsOrigin);
	var xsObj : object = xs;

	if (xs == xsOrigin)
	{
		raise "Return value xs should be a copied local object in clientMain";
	}

	if ((xsObj as (system::IRpcWrapperBase^) is not null))
	{
		raise "Return value xs should be a copied local object in clientMain";
	}

	var xsClient = xs;
	xs.Add(5);

	return
		$"[$(Print(xsOrigin))]" &
		$"[$(Print(xsService))]" &
		$"[$(Print(xsClient))]";
}
