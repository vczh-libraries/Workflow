module Rpc;
using system::*;

namespace RpcInheritanceMethodException
{
  @rpc:Interface
  interface IValue
  {
  }

  @rpc:Interface
  interface IOne : IValue
  {
    func SetOneValue() : void;
  }

  @rpc:Interface
  interface ITwo : IValue
  {
    func SetTwoValue() : void;
  }

  @rpc:Interface
  interface IDerived : IOne, ITwo
  {
  }

  @rpc:Interface
  @rpc:Ctor
  interface IService
  {
    func CreateDerived() : IDerived^;
  }
}
