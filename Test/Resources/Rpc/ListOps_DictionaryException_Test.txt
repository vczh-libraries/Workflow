module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcListOpsDictionaryException::*;

var s = "";

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (RpcListOpsDictionaryException::IService^)
  {
    override func DoSomethingWrong(xs : int[string]) : int
    {
      return xs["Anyway"];
    }
  };
  lc.RegisterLocalService(rpc_GetIds()["RpcListOpsDictionaryException::IService"], serviceObj);
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var service = cast (RpcListOpsDictionaryException::IService^) lc.RequestService("RpcListOpsDictionaryException::IService");
  try
  {
    service.DoSomethingWrong({});
  }
  catch (ex)
  {
    s = ex.Message;
  }
  return s;
}
