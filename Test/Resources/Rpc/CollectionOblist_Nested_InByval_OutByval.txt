module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollectionOblist
{
	namespace Nested
	{
		namespace InByval
		{
			namespace OutByval
			{
				@rpc:Interface
				@rpc:Ctor
				interface IService
				{
					@rpc:Byval
					func DoList(@rpc:Byval xs : observe (observe int[])[]) : observe (observe int[])[];
				}
			}
		}
	}
}

var xsService : observe (observe int[])[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (RpcCollectionOblist::Nested::InByval::OutByval::IService^)
	{
		override func DoList(xs : observe (observe int[])[]) : observe (observe int[])[]
		{
			var xsObj : object = xs;
			if ((xsObj as (system::IRpcWrapperBase^) is not null))
			{
				raise "Parameter xs should be a copied local object in serviceMain";
			}

			xsService = xs;
			var ys = xs[0];
			ys.Add(4);
			return xs;
		}
	};
	lc.RegisterService("RpcCollectionOblist::Nested::InByval::OutByval::IService", serviceObj);
}

func Print(xs : observe (observe int[])[]) : string
{
	var result = "";
	for (x in xs[0])
	{
		result = $"$(result)$(x)";
	}
	return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin = {{1 2 3}};
	var service = cast (RpcCollectionOblist::Nested::InByval::OutByval::IService^) lc.RequestService("RpcCollectionOblist::Nested::InByval::OutByval::IService");
	var xs = service.DoList(xsOrigin);

	if (xs == xsOrigin)
	{
		raise "Return value xs should be a copied local object in clientMain";
	}

	var xsObj : object = xs;

	if ((xsObj as (system::IRpcWrapperBase^) is not null))
	{
		raise "Return value xs should be a copied local object in clientMain";
	}

	var xsClient = xs;
	var ys = xs[0];
	ys.Add(5);

	return
		$"[$(Print(xsOrigin))]" &
		$"[$(Print(xsService))]" &
		$"[$(Print(xsClient))]";
}
