module Rpc;
using system::*;
using RpcEventOblist::*;

namespace RpcEventOblist
{
  @rpc:Interface
  @rpc:Ctor
  interface IService
  {
    func SetList(xs : observe int[]) : void;
    func ModifyHeldList() : void;
    func WatchHeldList() : void;
  }
}
var s = "";

func RecordItemChanged(handler : string, index : int, oldCount : int, newCount : int) : void
{
  s = $"$(s)[$(handler):$(index),$(oldCount),$(newCount)]";
}

func ServiceItemChanged(index : int, oldCount : int, newCount : int) : void
{
  RecordItemChanged("serviceMain", index, oldCount, newCount);
}

func ClientItemChanged(index : int, oldCount : int, newCount : int) : void
{
  RecordItemChanged("clientMain", index, oldCount, newCount);
}

func Modify(xs : observe int[]) : void
{
  xs.Add(10);
  xs.Insert(1, 20);
  xs.Set(0, 30);
  xs.RemoveAt(1);
  xs.Clear();
}

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (RpcEventOblist::IService^)
  {
    var _List : observe int[] = null;

    override func SetList(xs : observe int[]) : void
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is null))
      {
        raise "Parameter xs should be a wrapper object in serviceMain";
      }

      _List = xs;
    }

    override func ModifyHeldList() : void
    {
      Modify(_List);
    }

    override func WatchHeldList() : void
    {
      attach(_List.ItemChanged, ServiceItemChanged);
    }
  };
  lc.RegisterService("RpcEventOblist::IService", serviceObj);
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var xs : observe int[] = {};
  var service = cast (RpcEventOblist::IService^) lc.RequestService("RpcEventOblist::IService");
  service.SetList(xs);

  var handler = attach(xs.ItemChanged, ClientItemChanged);
  service.ModifyHeldList();
  detach(xs.ItemChanged, handler);

  service.WatchHeldList();
  Modify(xs);

  return s;
}
