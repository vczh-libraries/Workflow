module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollectionDict
{
	namespace InByval
	{
		namespace OutByval
		{
			@rpc:Interface
			@rpc:Ctor
			interface IService
			{
				@rpc:Byval
				func DoList(@rpc:Byval xs : string[int]) : string[int];
			}
		}
	}
}

var xsService : string[int] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (RpcCollectionDict::InByval::OutByval::IService^)
	{
		override func DoList(xs : string[int]) : string[int]
		{
			var xsObj : object = xs;
			if ((xsObj as (system::IRpcWrapperBase^) is not null))
			{
				raise "Parameter xs should be a copied local object in serviceMain";
			}

			xsService = xs;
			xs.Set(4, "D");
			return xs;
		}
	};
	lc.RegisterService("RpcCollectionDict::InByval::OutByval::IService", serviceObj);
}

func Print3(xs : string[int]) : string
{
	return $"1$(xs[1])2$(xs[2])3$(xs[3])";
}

func Print4(xs : string[int]) : string
{
	return $"1$(xs[1])2$(xs[2])3$(xs[3])4$(xs[4])";
}

func Print5(xs : string[int]) : string
{
	return $"1$(xs[1])2$(xs[2])3$(xs[3])4$(xs[4])5$(xs[5])";
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin = {1:"A" 2:"B" 3:"C"};
	var service = cast (RpcCollectionDict::InByval::OutByval::IService^) lc.RequestService("RpcCollectionDict::InByval::OutByval::IService");
	var xs = service.DoList(xsOrigin);
	var xsObj : object = xs;

	if ((xsObj as (system::IRpcWrapperBase^) is not null))
	{
		raise "Return value xs should be a copied local object in clientMain";
	}

	var xsClient = xs;
	xs.Set(5, "E");

	return
		$"[$(Print3(xsOrigin))]" &
		$"[$(Print4(xsService))]" &
		$"[$(Print5(xsClient))]";
}
