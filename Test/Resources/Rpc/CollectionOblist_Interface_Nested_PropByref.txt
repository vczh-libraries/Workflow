namespace RpcCollectionOblist

  namespace Interface

    namespace Nested
    {
      namespace InByref
      {
        namespace PropByref
        {
          @rpc:Interface
          interface IValue

            prop Value : int {GetValue}

          @rpc:Interface
          @rpc:Ctor
          interface IService
          {
            @rpc:Byref

            prop List : (observe IValue^[])[int] {not observe}

        }
      }
    }
  }
