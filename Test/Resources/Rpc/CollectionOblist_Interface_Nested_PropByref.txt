module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcCollectionOblist::Interface::Nested::InByref::PropByref::*;

namespace RpcCollectionOblist
{
  namespace Interface
  {
    namespace Nested
    {
      namespace InByref
      {
        namespace PropByref
        {
          @rpc:Interface
          interface IValue
          {
            func GetValue() : int;
            @rpc:Dynamic
            prop Value : int {GetValue}
          }

          @rpc:Interface
          @rpc:Ctor
          interface IService
          {
            @rpc:Byref
            @rpc:Dynamic
            prop List : (observe IValue^[])[int] {not observe}
          }
        }
      }
    }
  }
}

var xsService : (observe IValue^[])[int] = {};

func MakeValue(value : int) : IValue^
{
  return new (IValue^)
  {
    override func GetValue() : int
    {
      return value;
    }
  };
}

func CheckValue(value : IValue^, shouldBeWrapper : bool, name : string) : void
{
  var valueObj : object = value;
  if (shouldBeWrapper)
  {
    if ((valueObj as (system::IRpcWrapperBase^) is null))
    {
      raise $"$(name) should be a wrapper object";
    }
  }
  else
  {
    if ((valueObj as (system::IRpcWrapperBase^) is not null))
    {
      raise $"$(name) should be a local object";
    }
  }
}

func CheckServiceItems(xs : (observe IValue^[])[int]) : void
{
  var ys = xs[0];
  for (i in range [0, ys.Count))
  {
    CheckValue(ys[i], i != 3, $"xsService[0][$(i)]");
  }
}

func CheckClientItems(xs : (observe IValue^[])[int]) : void
{
  var ys = xs[0];
  for (i in range [0, ys.Count))
  {
    CheckValue(ys[i], i == 3, $"xsClient[0][$(i)]");
  }
}

func serviceMain(lc : IRpcLifeCycle*) : void
{
  var serviceObj = new (IService^)
  {
    var _List : (observe IValue^[])[int] = null;

    override func GetList() : (observe IValue^[])[int]
    {
      return _List;
    }

    override func SetList(xs : (observe IValue^[])[int]) : void
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is null))
      {
        raise "Parameter xs should be a wrapper object in serviceMain";
      }

      xsService = xs;
      var ys = xs[0];
      ys.Add(MakeValue(4));
      _List = xs;
    }
  };
  lc.RegisterService("RpcCollectionOblist::Interface::Nested::InByref::PropByref::IService", serviceObj);
}

func Print(xs : (observe IValue^[])[int]) : string
{
  var result = "";
  for (x in xs[0])
  {
    result = $"$(result)$(x.Value)";
  }
  return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
  var ysOrigin : observe IValue^[] = {};
  ysOrigin.Add(MakeValue(1));
  ysOrigin.Add(MakeValue(2));
  ysOrigin.Add(MakeValue(3));
  var xsOrigin : (observe IValue^[])[int] = {};
  xsOrigin.Set(0, ysOrigin);
  var service = cast (IService^) lc.RequestService("RpcCollectionOblist::Interface::Nested::InByref::PropByref::IService");
  service.List = xsOrigin;
  var xs = service.List;

  if (xs != xsOrigin)
  {
    raise "Return value xs should be the original parameter object in clientMain";
  }

  var xsClient = xs;
  var ys = xs[0];
  ys.Add(MakeValue(5));

  CheckServiceItems(xsService);
  CheckClientItems(xsClient);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
