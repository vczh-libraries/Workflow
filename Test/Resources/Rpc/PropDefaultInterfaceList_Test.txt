module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcPropDefaultInterfaceList::*;


var s = "";

func ClientValueChanged() : void
{
  s = $"$(s)[ValueChanged]";
}

func MakeValue(value : string) : IValue^
{
  return new (IValue^)
  {
    override func GetValue() : string
    {
      return value;
    }
  };
}

func Print(xs : IValue^[]) : string
{
  var result = "";
  for (x in xs)
  {
    result = $"$(result)$(x.Value)";
  }
  return result;
}

func CheckValue(value : IValue^, shouldBeWrapper : bool, name : string) : void
{
  var valueObj : object = value;
  if (shouldBeWrapper)
  {
    if ((valueObj as (system::IRpcWrapperBase^) is null))
    {
      raise $"$(name) should be a wrapper object";
    }
  }
  else
  {
    if ((valueObj as (system::IRpcWrapperBase^) is not null))
    {
      raise $"$(name) should be a local object";
    }
  }
}

func CheckList(xs : IValue^[], shouldBeWrapper : bool, name : string) : void
{
  var xsObj : object = xs;
  if (shouldBeWrapper)
  {
    if ((xsObj as (system::IRpcWrapperBase^) is null))
    {
      raise $"$(name) should be a wrapper object";
    }
  }
  else
  {
    if ((xsObj as (system::IRpcWrapperBase^) is not null))
    {
      raise $"$(name) should be a local object";
    }
  }

  for (i in range [0, xs.Count))
  {
    CheckValue(xs[i], shouldBeWrapper, $"$(name)[$(i)]");
  }
}

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (IService^)
  {
    var _Value : IValue^[] = {};

    override func GetValue() : IValue^[]
    {
      return _Value;
    }

    override func SetValue(value : IValue^[]) : void
    {
      CheckList(value, true, "Parameter value in serviceMain");
      _Value = value;
    }

    override func Signal() : void
    {
      ValueChanged();
    }
  };
  serviceObj.Value.Add(MakeValue("A"));
  lc.RegisterLocalService(rpc_GetIds()["RpcPropDefaultInterfaceList::IService"], serviceObj);
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var service = cast (IService^) lc.RequestService("RpcPropDefaultInterfaceList::IService");
  attach(service.ValueChanged, ClientValueChanged);

  var valueA = service.Value;
  CheckList(valueA, true, "service.Value before SetValue");
  s = $"$(s)[$(Print(valueA))]";

  var valueB : IValue^[] = {};
  valueB.Add(MakeValue("B"));
  valueB.Add(MakeValue("C"));
  valueB.Add(MakeValue("D"));
  CheckList(valueB, false, "valueB");
  service.SetValue(valueB);

  var valueCached = service.Value;
  if (valueCached != valueA)
  {
    raise "Cached value should be the original wrapper object in clientMain";
  }
  CheckList(valueCached, true, "service.Value before ValueChanged");
  s = $"$(s)[$(Print(valueCached))]";

  service.Signal();
  var value = service.Value;
  if (value != valueB)
  {
    raise "Value should be the original local object in clientMain";
  }
  CheckList(value, false, "service.Value after ValueChanged");
  s = $"$(s)[$(Print(value))]";

  return s;
}
