module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollectionDict
{
	namespace Nested
	{
		namespace InByval
		{
			namespace OutByref
			{
				@rpc:Interface
				@rpc:Ctor
				interface IService
				{
					@rpc:Byref
					func DoList(@rpc:Byval xs : (string[int])[]) : (string[int])[];
				}
			}
		}
	}
}

var xsService : (string[int])[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (RpcCollectionDict::Nested::InByval::OutByref::IService^)
	{
		override func DoList(xs : (string[int])[]) : (string[int])[]
		{
			var xsObj : object = xs;
			if ((xsObj as (system::IRpcWrapperBase^) is not null))
			{
				raise "Parameter xs should be a copied local object in serviceMain";
			}

			xsService = xs;
			var ys = xs[0];
			ys.Set(4, "D");
			return xs;
		}
	};
	lc.RegisterService("RpcCollectionDict::Nested::InByval::OutByref::IService", serviceObj);
}

func Print(xs : (string[int])[]) : string
{
	var ys = xs[0];
	var result = "";
	for (key in range [1, ys.Count])
	{
		result = $"$(result)$(key)$(ys[key])";
	}
	return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin = {{1:"A" 2:"B" 3:"C"}};
	var service = cast (RpcCollectionDict::Nested::InByval::OutByref::IService^) lc.RequestService("RpcCollectionDict::Nested::InByval::OutByref::IService");
	var xs = service.DoList(xsOrigin);
	var xsObj : object = xs;

	if ((xsObj as (system::IRpcWrapperBase^) is null))
	{
		raise "Return value xs should be a wrapper object in clientMain";
	}

	var xsClient = xs;
	var ys = xs[0];
	ys.Set(5, "E");

	return
		$"[$(Print(xsOrigin))]" &
		$"[$(Print(xsService))]" &
		$"[$(Print(xsClient))]";
}
