module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollectionOblist
{
	namespace Nested
	{
		namespace InByref
		{
			namespace PropByref
			{
				@rpc:Interface
				@rpc:Ctor
				interface IService
				{
					@rpc:Byref
					prop List : (int[])[] {not observe}
				}
			}
		}
	}
}

var xsService : (int[])[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (RpcCollectionOblist::Nested::InByref::PropByref::IService^)
	{
		var _List : (int[])[] = null;

		override func GetList() : (int[])[]
		{
			return _List;
		}

		override func SetList(xs : (int[])[]) : void
		{
			var xsObj : object = xs;
			if ((xsObj as (system::IRpcWrapperBase^) is null))
			{
				raise "Parameter xs should be a wrapper object in serviceMain";
			}

			xsService = xs;
			var ys = xs[0];
			ys.Add(4);
			_List = xs;
		}
	};
	lc.RegisterService("RpcCollectionOblist::Nested::InByref::PropByref::IService", serviceObj);
}

func Print(xs : (int[])[]) : string
{
	var result = "";
	for (x in xs[0])
	{
		result = $"$(result)$(x)";
	}
	return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin = {{1 2 3}};
	var service = cast (RpcCollectionOblist::Nested::InByref::PropByref::IService^) lc.RequestService("RpcCollectionOblist::Nested::InByref::PropByref::IService");
	service.List = xsOrigin;
	var xs = service.List;

	if (xs != xsOrigin)
	{
		raise "Return value xs should be the original parameter object in clientMain";
	}

	var xsClient = xs;
	var ys = xs[0];
	ys.Add(5);

	return
		$"[$(Print(xsOrigin))]" &
		$"[$(Print(xsService))]" &
		$"[$(Print(xsClient))]";
}
