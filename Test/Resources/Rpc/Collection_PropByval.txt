module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollection
{
  namespace InByval
  {
    namespace PropByval
    {
      @rpc:Interface
      @rpc:Ctor
      interface IService
      {
        @rpc:Byval
        @rpc:Dynamic
        prop List : int[] {not observe}
      }
    }
  }
}

var xsService : int[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
  var serviceObj = new (RpcCollection::InByval::PropByval::IService^)
  {
    var _List : int[] = null;

    override func GetList() : int[]
    {
      return _List;
    }

    override func SetList(xs : int[]) : void
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is not null))
      {
        raise "Parameter xs should be a copied local object in serviceMain";
      }

      xsService = xs;
      xs.Add(4);
      _List = xs;
    }
  };
  lc.RegisterService("RpcCollection::InByval::PropByval::IService", serviceObj);
}

func Print(xs : int[]) : string
{
  var result = "";
  for (x in xs)
  {
    result = $"$(result)$(x)";
  }
  return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
  var xsOrigin = {1 2 3};
  var service = cast (RpcCollection::InByval::PropByval::IService^) lc.RequestService("RpcCollection::InByval::PropByval::IService");
  service.List = xsOrigin;
  var xs = service.List;
  var xsObj : object = xs;

  if (xs == xsOrigin)
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  if ((xsObj as (system::IRpcWrapperBase^) is not null))
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  var xsClient = xs;
  xs.Add(5);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
