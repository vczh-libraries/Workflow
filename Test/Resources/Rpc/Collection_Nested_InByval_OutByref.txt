module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollection
{
	namespace Nested
	{
		namespace InByval
		{
			namespace OutByref
			{
				@rpc:Interface
				@rpc:Ctor
				interface IService
				{
					@rpc:Byref
					func DoList(@rpc:Byval xs : (int[])[]) : (int[])[];
				}
			}
		}
	}
}

var xsService : (int[])[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (RpcCollection::Nested::InByval::OutByref::IService^)
	{
		override func DoList(xs : (int[])[]) : (int[])[]
		{
			var xsObj : object = xs;
			if ((xsObj as (system::IRpcWrapperBase^) is not null))
			{
				raise "Parameter xs should be a copied local object in serviceMain";
			}

			xsService = xs;
			var ys = xs[0];
			ys.Add(4);
			return xs;
		}
	};
	lc.RegisterService("RpcCollection::Nested::InByval::OutByref::IService", serviceObj);
}

func Print(xs : (int[])[]) : string
{
	var result = "";
	for (x in xs[0])
	{
		result = $"$(result)$(x)";
	}
	return result;
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin = {{1 2 3}};
	var service = cast (RpcCollection::Nested::InByval::OutByref::IService^) lc.RequestService("RpcCollection::Nested::InByval::OutByref::IService");
	var xs = service.DoList(xsOrigin);

	var xsClient = xs;
	var ys = xs[0];
	ys.Add(5);

	return
		$"[$(Print(xsOrigin))]" &
		$"[$(Print(xsService))]" &
		$"[$(Print(xsClient))]";
}
