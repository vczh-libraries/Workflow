module Rpc;
using system::*;
using RpcWrapperTest::*;

namespace RpcCollection
{
  namespace Nested
  {
    namespace Default
    {
      @rpc:Interface
      @rpc:Ctor
      interface IService
      {
        func DoList(xs : (int[])[]) : (int[])[];
      }
    }
  }
}

var xsService : (int[])[] = {};

func serviceMain(lc : IRpcLifecycle*) : void
{
  var serviceObj = new (RpcCollection::Nested::Default::IService^)
  {
    override func DoList(xs : (int[])[]) : (int[])[]
    {
      var xsObj : object = xs;
      if ((xsObj as (system::IRpcWrapperBase^) is not null))
      {
        raise "Parameter xs should be a copied local object in serviceMain";
      }

      xsService = xs;
      var ys = xs[0];
      ys.Add(4);
      return xs;
    }
  };
  lc.RegisterService("RpcCollection::Nested::Default::IService", serviceObj);
}

func Print(xs : (int[])[]) : string
{
  var result = "";
  for (x in xs[0])
  {
    result = $"$(result)$(x)";
  }
  return result;
}

func clientMain(lc : IRpcLifecycle*) : string
{
  var xsOrigin = {{1 2 3}};
  var service = cast (RpcCollection::Nested::Default::IService^) lc.RequestService("RpcCollection::Nested::Default::IService");
  var xs = service.DoList(xsOrigin);

  if (xs == xsOrigin)
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  var xsObj : object = xs;

  if ((xsObj as (system::IRpcWrapperBase^) is not null))
  {
    raise "Return value xs should be a copied local object in clientMain";
  }

  var xsClient = xs;
  var ys = xs[0];
  ys.Add(5);

  return
    $"[$(Print(xsOrigin))]" &
    $"[$(Print(xsService))]" &
    $"[$(Print(xsClient))]";
}
