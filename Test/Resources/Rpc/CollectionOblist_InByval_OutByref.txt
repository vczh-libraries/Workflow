module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcCollectionOblist::InByval::OutByref::*;

namespace RpcCollectionOblist
{
	namespace InByval
	{
		namespace OutByref
		{
			@rpc:Interface
			@rpc:Ctor
			interface IService
			{
				@rpc:Byref
				func DoList(@rpc:Byval xs : int[]) : int[];
			}
		}
	}
}

var xsService : int[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (IService^)
	{
		override func DoList(xs : int[]) : int[]
		{
			var xsObj : object = xs;
			if ((xsObj as (system::IRpcWrapperBase^) is not null))
			{
				raise "Parameter xs should be a copied local object in serviceMain";
			}

			xsService = xs;
			xs.Add(4);
			return xs;
		}
	};
	lc.RegisterService("RpcCollectionOblist::InByval::OutByref::IService", serviceObj);
}

func Print3(xs : int[]) : string
{
	return $"$(xs[0])$(xs[1])$(xs[2])";
}

func Print5(xs : int[]) : string
{
	return $"$(xs[0])$(xs[1])$(xs[2])$(xs[3])$(xs[4])";
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin = {1 2 3};
	var service = cast (IService^) lc.RequestService("RpcCollectionOblist::InByval::OutByref::IService");
	var xs = service.DoList(xsOrigin);
	var xsObj : object = xs;

	if ((xsObj as (system::IRpcWrapperBase^) is null))
	{
		raise "Return value xs should be a wrapper object in clientMain";
	}

	var xsClient = xs;
	xs.Add(5);

	return
		$"[$(Print3(xsOrigin))]" &
		$"[$(Print5(xsService))]" &
		$"[$(Print5(xsClient))]";
}
