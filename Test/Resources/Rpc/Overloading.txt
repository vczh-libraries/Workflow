module Rpc;
using system::*;
using RpcOverloadingTest::*;

namespace RpcOverloadingTest
{
  @rpc:Interface
  interface IStringRepresentable
  {
    func GetStringValue() : string;
    @rpc:Dynamic
    prop StringValue : string { GetStringValue }
  }

  @rpc:Interface
  @rpc:Ctor
  interface IService
  {
    func ToStringInt(value : int) : string;
    func ToString(value : bool) : string;
    func ToString(value : string) : string;
    func ToString(value : IStringRepresentable^) : string;
    func ToString(value1 : int, value2 : bool, value3 : string, value4 : IStringRepresentable^) : string;
  }
}
