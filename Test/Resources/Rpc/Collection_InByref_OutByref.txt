module Rpc;
using system::*;
using RpcWrapperTest::*;
using RpcCollection::InByref::OutByref::*;

namespace RpcCollection
{
	namespace InByref
	{
		namespace OutByref
		{
			@rpc:Interface
			@rpc:Ctor
			interface IService
			{
				@rpc:Byref
				func DoList(@rpc:Byref xs : int[]) : int[];
			}
		}
	}
}

var xsService : int[] = {};

func serviceMain(lc : IRpcLifeCycle*) : void
{
	var serviceObj = new (IService^)
	{
		override func DoList(xs : int[]) : int[]
		{
			xsService = xs;
			xs.Add(4);
			return xs;
		}
	};
	lc.RegisterService("RpcCollection::InByref::OutByref::IService", serviceObj);
}

func Print5(xs : int[]) : string
{
	return $"$(xs[0])$(xs[1])$(xs[2])$(xs[3])$(xs[4])";
}

func clientMain(lc : IRpcLifeCycle*) : string
{
	var xsOrigin = {1 2 3};
	var service = cast (IService^) lc.RequestService("RpcCollection::InByref::OutByref::IService");
	var xs = service.DoList(xsOrigin);

	if (xs != xsOrigin)
	{
		raise "Return value xs should be the original parameter object in clientMain";
	}

	var xsClient = xs;
	xs.Add(5);

	return
		$"[$(Print5(xsOrigin))]" &
		$"[$(Print5(xsService))]" &
		$"[$(Print5(xsClient))]";
}
